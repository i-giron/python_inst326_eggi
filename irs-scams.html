Defining the IRS Scams Manager Class¶

The manager class contains functions to prepare our cvs files for discoveries that will answer our research question.

In [2]:
# Isaac
class IRSScamsManager:
    
    """
    This class contains the functions necessary to prepare the CSV file to receive analysis treatment
    """
    
    # Eric
    # Regular expression pattern to be replaced
    name_replace_pat = re.compile(r"[^a-z]+")
    
    # The columns and columns names differed slightly between the
    # two Excel files, so we clean them up here.
    @staticmethod
    def convert_col_names(names):
        """
        This function looks at patterns to search so that we replace it in the column names
        
        Args:
            (list): list of column names
        """
        return [ re.sub(IRSScamsManager.name_replace_pat, "_", name.lower().strip())
            for name in names ]
    
    @staticmethod
    def rename_cols(df):
        """
        This function directly renames the column names
        
        Args:
            (dataframe): csv file
        """
        return df.rename(columns=dict(zip(df.columns, IRSScamsManager.convert_col_names(df.columns))))
    
    @staticmethod
    def clean_state_codes(state_name):
        """
        This function cleans the state codes by removing unneccesary characters
        
        Args:
            (string): state code
        """
        if type(state_name) == str and "|" in state_name:
            non_null = list(filter(lambda x: x != "", state_name.split("|")))
            if len(non_null):
                # Return first element of list, which is the state code name
                return non_null[0]
            else:
                return None
        else:
            # Return state as it is if it is already cleaned
            return state_name

root = Tk()

all_US_states = df['clean_state']
top_20_US_states_to_scam = all_US_states.dropna().value_counts().head(20)
top_20_US_states_names = top_20_US_states_to_scam.index.tolist()

# Create a Tkinter variable
tkvar = StringVar(root)
tkvar.set(sorted(top_20_US_states_names)[0]) # set the default option

popupMenu = OptionMenu(root, tkvar, *sorted(top_20_US_states_names))
popupMenu.pack()

def show_diagram():
    statesTotal ={'States': top_20_US_states_names, 'Total': top_20_US_states_to_scam}
    df1 = pd.DataFrame(statesTotal, columns = ['States', 'Total'])
    df1 = df1[['States', 'Total']].groupby('States').sum()

    figure1 = mpl.pyplot.Figure(figsize=(20, 2), dpi=80)
    ax1 = figure1.add_subplot(111)
    ax1.set_xlabel('\n\nStates')
    ax1.set_ylabel('Total\n\n\n')
    bar1 = FigureCanvasTkAgg(figure1, root)
    bar1.get_tk_widget().pack(side=LEFT, fill=BOTH)

    explode_largest = list()
    for idx in range(len(top_20_US_states_names)):
        if tkvar.get() == sorted(top_20_US_states_names)[idx]:
            explode_largest.append(0.1)
        else:
            explode_largest.append(0.0)
    
    df1.plot.pie(autopct='%.2f', explode=explode_largest, legend=False, 
                 subplots=True, ax=ax1, shadow=True, startangle=90)
    ax1.set_title('States Vs. Total Scams\n\n\n')

button1 = Button(root, text="Choose State", command=show_diagram)
button1.pack()

button2 = Button(root, text='Exit Application', command=root.destroy)
button2.pack()
